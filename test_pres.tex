\begin{frame}{Схема управления по прогнозирующей модели}
 \begin{block}{\begin{center} \textbf{Общая схема управления с предсказанием описывается следующим образом:}  \end{center}}
	\begin{enumerate}
	\item Оценивание  или измерение вектора состояния  реального объекта.
	\item Решение задачи (нахождение положения равновесия).
	\item Использование найденной оптимальной функции $u^*$ в качестве программного управления на отрезке $\tau \in [t, t+\delta]$.
	\item Замена момента времени $t$ на момент $t+\delta$ и повторение операций в пунктах 1-3.
	\end{enumerate}
  \end{block}
\end{frame}

\begin{frame}{Задача оптимизации неоклассической модели экономического роста}
\begin{block}{}
\begin{enumerate}
	\item Критерий качества
	\begin{equation}
	J(x,u) = \int_0^\infty e^{-\rho t} [\ln(1- u(t)) + \ln f(x(t))] dt \to \max \label{Ex1_J}
	\end{equation}
	\item Математическая модель
	\begin{equation}
	 \dot{x}(t) = u(t) f(x(t)) - \mu x(t),\quad x(0) = x_0 \label{Ex1_dotx}
	\end{equation}
	\item Ограничения на управления
	\begin{equation}
	0 \leqslant u \leqslant 1 - \varepsilon \label{Ex1_u}
	\end{equation}
\end{enumerate}
\end{block}

{\it где $0 < \varepsilon < 1, \; \mu > 0$ --- некоторые постоянные. \\
$ \quad \; \; \rho > 0$ --- параметр дисконтирования.}
\end{frame}

\begin{frame}{Задача оптимизации неоклассической модели экономического роста}
   \begin{block}{\begin{center} \textbf{}  \end{center}}
  \begin{align}
	J(x,u) = &\int_0^{\tau_N} w|x(t) - x^*|^2 + |u(t) - u^*|^2 dt \to \min, \label{Ex1_MPC_J} \\
	\dot{x}(t) &= u(t) f(x(t)) - \mu x(t), \quad x(0) = x_{\tau_p}, \label{Ex1_MPC_dotx}\\
	u_s f&(x_{\tau_N}) - \mu x_{\tau_N} = 0, \label{Ex1_MPC}\\
	 &0 \leqslant u(t) \leqslant 1-\varepsilon, \quad t \in [0, \tau_N). \label{Ex1_MPC_u}\\	
\end{align}
   \end{block}
{\it где $w$ --- весовой коэффициент.} \\
Уравнение (\ref{Ex1_MPC}) обеспечивает, что траектория предсказанного состояния $x(t)$ сходится к допустимому промежуточному состоянию.
\end{frame}
