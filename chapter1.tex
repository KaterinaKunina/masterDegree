\chapter{ЗАДАЧИ ОПТИМАЛЬНОГО УПРАВЛЕНИЯ НА БЕСКОНЕЧНОМ ПОЛУИНТЕРВАЛЕ ВРЕМЕНИ}\label{chap1}

Данный раздел посвящен обзору теории оптимального управления для класса задач на бесконечном полуинтервале времени, возникающих в экономике при исследовании динамических моделей оптимального распределения ресурсов \cite{Kryazhimski}. Задачи данного класса характеризуются отсутствием ограничений на поведение допустимых траекторий на бесконечности, фиксированным начальным состоянием, специальным видом интегрального функционала, задаваемого несобственным интегралом с параметром дисконтирования $\rho$.
Для решения задач оптимального управления со свободным временем формулируется в разделе \ref{chap1}.2 принцип максимума Понтрягина \cite{Kryazhimski}. Приведены условия, гарантирующие нормальность задачи и выполнение “стандартных” условий трансверсальности на бесконечности.

\section{Постановка задачи}
Будем рассматривать следующую задачу оптимального управления (P)

\begin{equation}\label{P1}
J(x,u)=\int_0^\infty e^{-\rho t} g(x(t),u(t)) dt \to \max,
\end{equation}

\begin{equation}\label{P2}
\dot{x} = f(x(t),u(t)), \quad u(t) \in U,
\end{equation}

\begin{equation}\label{P3}
x(0) = x_0.
\end{equation}

Здесь $x(t) = (x^1(t), \dots, x^n(t)) \in \R^n $ и $u(t)=(u^1(t), \dots, u^m(t)) \in \R^m$ -- значения в момент времени $t \geqslant 0$ 
фазового вектора управляемой системы (\ref{P2}) и ее вектора
управления соответственно; $U$ — непустой компакт в $\R^m$; $x_0 \in \R^n$ -- заданное начальное состояние системы;
$\rho \geqslant 0$ — параметр дисконтирования. Предполагается, что $x_0 \in G$, где $G$ — заданное открытое множество в $\R^n$. 

Далее предполагается, что векторная функция $f: G \times U \to \R^n$, скалярная функция $g: G \times U \to \R$, матричная функция 
$\frac{\partial f}{\partial x}$ и градиент 
$\frac{\partial g}{\partial x}$ непрерывны на декартовом произведении $G \times U$. 

Класс допустимых управлений системы (\ref{P1}) состоит из всех измеримых векторных функций $u: [0, \infty) \to U$. Допустимая траектория системы (\ref{P2}), соответствующая допустимому управлению $u$, понимается как определенное на $[0, \infty]$ решение системы (\ref{P2}), удовлетворяющее начальному условию (\ref{P3}). В дальнейшем считаем, что для любого допустимого управления существует соответствующая ему траектория, причем для любого $t \geqslant 0$ ее значения $x(t)$ лежат в $G$. В силу непрерывной дифференцируемости векторной функции $f$ по первой переменной $x$ такая траектория единственная. 

\newtheorem{Def}{Определение} [chapter]
\begin{Def}
 Пара $(x, u)$, где $u$ — допустимое управление и $x$ — соответствующая ему траектория, называется допустимой парой.
\end{Def}
 
Предполагается, что для любой допустимой пары $(x, u)$ интеграл в (\ref{P1}) сходится абсолютно.
 
\newtheorem{Def1}{Определение} [chapter]
\begin{Def}
	Допустимая пара $(x^0, u^0)$ называется оптимальной, если функционал полезности (\ref{P1}) принимает на ней наибольшее возможное значение.
\end{Def}

В этом случае $u^0$ — оптимальное допустимое управление, а $x^0$ — оптимальная траектория. 

В дальнейшем будем считать выполненными следующие условия. 

(A1) Существует такое $C_0 \geqslant 0$, что 
\[
	\langle x, f(x, u)\rangle \leqslant C_0(1 + \|x\|^2) \quad \forall x \in G, u \in U.
\]

(A2) Для всякого $x \in G$ выпуклое множество 
\[
	Q(x)= \{(z^0,z) \in R^{n+1}: z^0 \leqslant g(x,u), z=f(x,u), u \in U\}.
\]

(A3) Существуют такие положительные функции $\mu$ и $w$ на $[0,+ \infty)$, что $\mu(t) \to +0, w(t) \to +0$ при $t \to + \infty$ и, какова быни была допустимая пара $(x, u)$, выполняются неравенства

\begin{equation} \label{A3_1} 
	e^{-\rho t} \max_{u \in U} |g(x(t),u)| \leqslant \mu(t) \quad \forall t \geqslant 0,
\end{equation}

\begin{equation} \label{A3_2}
	\int_T^\infty e^{-\rho t} |g(x(t),u(t))|dt \leqslant w(T) \quad \forall T \geqslant 0.
\end{equation}

Как уже отмечалось выше, основная трудность исследования задачи (\ref{P1})–(\ref{P3}) состоит в том, что процесс управления системой (\ref{P2}) рассматривается на бесконечном полуинтервале времени. Эта трудность связана с тем обстоятельством, что наличие бесконечного полуинтервала вносит в задачу особенность.

Для иллюстрации этого эффекта в случае $\rho > 0$ сделаем в задаче замену времени

\begin{equation}\label{new_time(8)}
	\tau(t)=1-e^{-\rho t}, \quad t \in [0, + \infty].
\end{equation}

Непрерывно дифференцируемая функция $\tau : [0, + \infty) \to [0, 1)$ монотонно возрастает. Следовательно, замена (\ref{new_time(8)}) приводит к следующей задаче оптимального управления на конечном полуинтервале времени [0, 1):

\begin{equation}\label{Problem_with_timeEnd}
 \begin{split}
	J(x,u) = \int_0^1 g&(x(\tau),u(\tau))d \tau \to \max, \\
	\frac{d}{d \tau} x(\tau) = \frac{1}{\rho(1-\tau)}&f(x(\tau),u(\tau)), \quad u(\tau) \in U, \\
	&x(0) = x_0.
 \end{split}
\end{equation}

Здесь класс допустимых управлений состоит из всех измеримых функций $u$: $[0, 1) \to U$. 

В задаче (\ref{Problem_with_timeEnd}) имеется особенность в конечный момент $\tau_1 = 1$. В силу этой особенности правая часть управляемой системы (\ref{Problem_with_timeEnd}) может стремиться к бесконечности при $\tau \to 1$.

В моделях экономики координаты фазового вектора $x(t)$ часто трактуются как величины различных производственных факторов, участвующих в процессе производства в текущий момент времени $t \geqslant 0$, таких, как капитал (основные производственные фонды) $K(t)$, материалы $M(t)$, трудовые ресурсы $L(t)$ и т.д. При этом вектор $u(t)$ управления обычно характеризует текущие размеры инвестиций в производственные факторы. Значения подынтегральной функции $g(x(t), u(t))$, определяющие после интегрирования с дисконтированием итоговую величину показателя полезности $J(x, u)$, обычно характеризуют текущую мгновенную полезность допустимой пары $(x, u)$. 

Задачи оптимального управления в экономике могут формулироваться как на микроуровне (фирма или предприятие), так и на макроуровне (отрасль, государство или вся мировая экономика). Для моделей предприятия в качестве мгновенной полезности $g(x(t), u(t))$ часто рассматривают прибыль, получаемую предприятием в единицу времени, следующую за моментом времени $t \geqslant 0$ (текущую прибыль). В макроэкономических моделях оптимизации экономического роста мгновенная полезность может определяться величиной потребления на душу населения в единицу времени, величиной прироста потребления на душу населения в единицу времени и др. В микроэкономических моделях дисконтирующий множитель $e^{-\rho t}$ в интеграле (\ref{P1}) иногда связывают с величиной инфляции — удельной скоростью обесценивания “реальной” стоимости используемой денежной единицы.

\section{Принцип максимума и условия трансверсальности на бесконечности} %\label{chap2}

%В данном разделе формулируется принцип максимума Понтрягина для задачи ($P$).

Определим функцию Гамильтона-Понтрягина $\mathcal{H}: G \times [0, \infty] \times U \times \R^n \times \R^v \to \R^v$ и гамильтониан $H: G \times [0, \infty) \times \R^n \times \R^v \to \R^v$ для задачи ($P$) стандартным образом:

\[
	\mathcal{H}(x,t,u,\psi, \psi^0) = \langle f(x,u), \psi \rangle + \psi^0 e^{-\rho t} g(x,u), 
\]

\[
	H(s, t, \psi, \psi^0) = \sup_{u \in U} \mathcal{H}(x, t, u, \psi, \psi^0).
\]

Формулировка принципа максимума Понтрягина включает в себя допустимую пару ($x^0, u^0$) и пару ($\psi, \psi^0$) сопряженных переменных (или множителей Лагранжа), соответствующую
допустимой паре ($x^0, u^0$). Сопряженная переменная $\psi$ определяется как (локально) абсолютно непрерывная векторная функция на $[0, \infty)$ со значениями в $\R^n$, являющаяся решением на $[0, \infty)$ следующей линейной системыдифференциальных уравнений (сопряженной системы):

\begin{multline} \label{conjugate_system}
	\dot{\psi} (t) = - \frac{\partial \mathcal{H}(x^0(t),t,u^0(t),\psi (t),\psi^0)}{\partial x} = \\
	= -\Biggl[ \frac{\partial f(x^0(t), u^0(t))}{\partial x} \Biggr]^0 \psi (t) - \psi^0 e^{-\rho t} \frac{\partial g (x^0(t), u^0(t))}{\partial x}.
\end{multline}

Сопряженная переменная $\psi^0$ есть неотрицательное действительное число. Пара ($\psi, \psi^0$) сопряженных переменных называется нетривиальной, если

\begin{equation} \label{conjugate_psi}
	\| \psi(0)\| + \psi^0 > 0.
\end{equation}

Будем говорить, что допустимая пара ($x^0, u^0$) удовлетворяет основным соотношениям принципа максимума Понтрягина (для задачи (\ref{P1})–(\ref{P3})) вместе с соответствующей ей парой ($\psi, \psi^0$)
сопряженных переменных, если пара ($\psi, \psi^0$) является нетривиальной и на $[0,\infty)$ выполняется следующее условие максимума:

\begin{equation} \label{condition_max}
	\mathcal{H}(x^0(t), t, u^0(t), \psi (t), \psi^0) \overset{\text{п.в.}}{=} H(x^0(t), t, \psi(t), \psi^0).
\end{equation}

Принцип максимума Понтрягина наиболее информативен в случае, когда задача (\ref{P1})–(\ref{P3}) является нормальной, т.е. когда множитель Лагранжа $\psi^0$, фигурирующий в соотношениях (\ref{conjugate_system})–(\ref{condition_max}), отличен от нуля. В этом случае, не ограничивая общности, можно считать, что $\psi^0 = 1$. 

Определим функцию Гамильтона–Понтрягина в нормальной форме $\mathcal{H}: G \times [0, \infty) \times U \times \R^n \to \R^v$ и нормальную форму гамильтониана $H: G \times [0, \infty) \times \R^n \to \R^v$ следующим образом:

\[
	\mathcal{H}(x, t, u, \psi) = \mathcal{H}(x, t, u, \psi, 1) = \langle f(x, u), \psi \rangle + e^{-\rho t} g(x, u),
\]

\[
	H(x, t, \psi) = H(x, t, \psi, 1) = \sup_{u \in U} \mathcal{H}(x, t, u, \psi).
\]

Соответственно для всякой допустимой пары ($x^0, u^0$) введем нормальную форму сопряженной системы как систему дифференциальных уравнений

\begin{multline} \label{nomal_conjugate_system}
	\dot{\psi}(t) = - \frac{\partial H(x^0(t), t, u^0(t), \psi (t))}{\partial x} = \\
	= - \Biggl[ \frac{\partial f(x^0(t), u^0(t))}{\partial x} \Biggr]^0 \psi (t) - e^{-\rho t} \frac{\partial g(x^0(t), u^0(t))}{\partial x}.
\end{multline}

\newtheorem{Def2}{Определение} [chapter]
\begin{Def}
Любое определенное на $[0, + \infty)$ решение $\psi$ системы (\ref{nomal_conjugate_system}) будем называть сопряженной переменной (множителем Лагранжа), соответствующей допустимой паре ($x^0, u^0$). 
\end{Def}

Будем говорить, что допустимая пара ($x^0, u^0$) удовлетворяет основным соотношениям принципа максимума Понтрягина в нормальной форме вместе с соответствующей ей сопряженной переменной $\psi$, если на полуинтервале $[0, \infty)$ выполняется следующее условие максимума в нормальной форме:

\begin{equation} \label{normal_condition_max}
	\mathcal{H}(x^0(t), t, u^0(t), \psi (t)) \overset{\text{п.в.}}{=} H(x^0(t), t, \psi(t)).
\end{equation}

Заметим, что в этом случае условие (\ref{conjugate_psi}) нетривиальности выполняется автоматически, поскольку $\psi^0 = 1$.

\newtheorem{Th}{Теорема}
\begin{Th}[Принцип максимума]\label{thMaxPontragin}
	Пусть выполняются условия (A1)–(A3) и допустимая пара ($x^0, u^0$) оптимальна в задаче (P). Тогда пара ($x^0, u^0$) удовлетворяет основным соотношениям (\ref{conjugate_system})–(\ref{condition_max}) принципа максимума Понтрягина вместе с некоторой соответствующей ей парой ($\psi, \psi^0$) сопряженных переменных.
\end{Th}

Заметим, что формулировка теоремы \ref{thMaxPontragin} слабее формулировки соответствующего классического результата, известного для задачи оптимального управления на конечном интервале времени со свободным правым концом. 

%Для задачи (P) оптимального управления на бесконечном полуинтервале времени $[0, \infty)$ вопрос о получении дополнительных условий, характеризующих поведение сопряженной переменной на бесконечности.
В качестве условий трансверсальности на бесконечности обычно предлагается использовать:

\begin{equation} \label{transversality_condition1}
	\psi^0 = 1, \quad \lim_{t \to \infty} \psi (t) = 0
\end{equation}

или

\begin{equation} \label{transversality_condition2}
		\psi^0 = 1, \quad \lim_{t \to \infty} \langle x^0(t), \psi(t) \rangle = 0.
\end{equation}

Заметим, что использование условий (\ref{transversality_condition2}) в качестве необходимых условий оптимальности иногда мотивируется тем, что эти условия фигурируют в известных (основанных на соотношениях принципа максимума) достаточных условиях оптимальности для задачи (\ref{P1}) - (\ref{P3}) \cite{Kryazhimski}.

В общем случае для оптимальной допустимой пары ($x^0, u^0$) в задаче (\ref{P1}) - (\ref{P3}) и соответствующей ей пары сопряженных переменных ($\psi, \psi^0$) естественные условия трансверсальности (\ref{transversality_condition1})
или (\ref{transversality_condition2}) не выполняются. Более того, могут нарушаться оба условия как в соотношениях (\ref{transversality_condition1}), так и в (\ref{transversality_condition2}), причем в различных комбинациях \cite{Kryazhimski}.

%В следующем разделе, для задачи (\ref{P1}) - (\ref{P3}) с положительным параметром дисконтирования $\rho > 0$ мы приведем набор примеров, иллюстрирующих разнообразные патологии поведения сопряженных переменных на бесконечности.\\

Важной особенностью задачи (\ref{P1})–(\ref{P3}) оптимального управления на бесконечном полуинтервале времени является тот факт, что основные соотношения (\ref{conjugate_system})–(\ref{condition_max}) принципа максимума всегда можно дополнить следующим условием \textit{асимптотической стационарности} гамильтониана на бесконечности:

\begin{equation} \label{asimptotic_stationary_hamiltonian}
	\lim_{t \to \infty} H(x^0(t), t, \psi (t), \psi^0) = 0.
\end{equation}

Условие (\ref{asimptotic_stationary_hamiltonian}) асимптотической стационарности аналогично условию трансверсальности по времени, входящему в набор соотношений принципа максимума Понтрягина для задач оптимального управления со свободным(конечным) временем. Легко видеть, что в рамках задачи (\ref{P1})–(\ref{P3}) условие (\ref{asimptotic_stationary_hamiltonian}) эквивалентно следующему условию стационарности гамильтониана:

\begin{equation} \label{stationary_hamiltonian}
  \begin{split}
	&H(x^0 (t), t, \psi (t), \psi^0) = \rho \psi^0 \int_t^\infty e^{-\rho s} g(x^0 (x), u^0 (s)) ds \\
  &\forall t \geqslant 0.
 \end{split}
\end{equation}

Действительно, в силу условия (\ref{inada_condition3}) равенство (\ref{asimptotic_stationary_hamiltonian}) вытекает из (\ref{stationary_hamiltonian}). С другой стороны, при выполнении условия (\ref{A3_2}) условие (\ref{stationary_hamiltonian}) следует из равенства (\ref{asimptotic_stationary_hamiltonian}) и соотношения

\[
	\frac{d}{dt} H(x^0 (t), t, \psi (t), \psi^0) \overset{\text{п.в.}}{=} \frac{\partial}{\partial t} \mathcal{H} (x^0 (t), t, u^0 (t), \psi(t), \psi^0),
\]
выполнение которого на полуинтервале $[0, + \infty)$ в свою очередь вытекает из уравнений сопряженной системы (\ref{conjugate_system}) и условия максимума (\ref{condition_max}). Таким образом, в рамках задачи (\ref{P1})-(\ref{P3}) условия (\ref{asimptotic_stationary_hamiltonian}) и (\ref{stationary_hamiltonian}) эквивалентны. 

Заметим, что так как факт эквивалентности условий (\ref{asimptotic_stationary_hamiltonian}) и (\ref{stationary_hamiltonian}) не связан со свойством оптимальности допустимой пары ($x^0, u^0$), а вытекает из выполнения для этой пары соотношений принципа максимума Понтрягина, то справедливо следующее утверждение.

\newtheorem{Lem}[Th]{Лемма}
\begin{Lem} \label{Lem_compare_utility}
	Пусть $\rho > 0$ и ($x^0, u^0$) — допустимая пара системы (\ref{P2}), удовлетворяющая соотношениям (\ref{nomal_conjugate_system}), (\ref{normal_condition_max}) принципа максимума Понтрягина в нормальной форме с соответствующей ей сопряженной переменной $\psi$. Пусть, кроме того, интеграл

\[
	\int_0^\infty e^{-\rho s} g(x^0 (s), u^0(s))ds	
\]
сходится абсолютно (в задаче ($P$) это всегда так) и выполняется условие стационарности (\ref{asimptotic_stationary_hamiltonian}) (при $\psi^0 = 1$). Тогда имеем

\begin{equation} \label{lemma}
	\int_t^\infty e^{-\rho s} g(x^0(s), u^0(s)) ds = \frac{1}{\rho} H(x^0(t), t, \psi (t)) \quad \forall t \geqslant 0.
\end{equation}

\end{Lem}

Равенство (\ref{lemma}) позволяет производить прямое сравнение значений функционала полезности (\ref{P1}) на различных допустимых парах системы (\ref{P2}), удовлетворяющих условиям леммы \ref{Lem_compare_utility}.
